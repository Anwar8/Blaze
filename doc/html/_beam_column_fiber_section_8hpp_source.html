115 void increment_fibre_strains()
116 {
117 real d_axial_strain = axial_strain - starting_axial_strain;
118 real d_curvature = curvature - starting_curvature;
119
120 // careful - since each fibre has a unique_ptr, it cannot be copied and so must be accessed by reference even in the loop.
121 for (auto& fibre : fibres)
122 {
123 real strain_increment = d_axial_strain - (fibre.get_y() - y_bar)*d_curvature; // the minus sign is as per Izzuddin's notation.
124 fibre.material_ptr->increment_strain(strain_increment);
125 }

149 void calc_area_weighted_E()
150 {
151 section_area = 0.0;
152 real area_times_E = 0.0;
153 for (auto& fibre : fibres)
154 {
155 real fibre_area = fibre.get_area();
156 section_area += fibre_area;
157 area_times_E += fibre_area*fibre.material_ptr->get_E_t();
158 }
159 weighted_E = area_times_E/section_area;

166 void calc_section_centroid()
167 {
168 real area_moment = 0.0;
169 for (auto& fibre : fibres)
170 {
171 area_moment += fibre.get_y()*fibre.get_area()*(fibre.material_ptr->get_E_t());
172 }
173 y_bar = area_moment/(section_area*weighted_E);

191 void calc_tan_contitutive_matrix()
192 {
193 D_t.setZero();
194 for (auto& fibre : fibres)
195 {
196 real E_t_i = fibre.material_ptr->get_E_t();
197 real A_i = fibre.get_area();
198 real y_i = fibre.get_y();
199
200 D_t(0,0) = D_t(0,0) + A_i * E_t_i;
201 D_t(1,1) = D_t(1,1) + A_i * E_t_i * pow(y_i - y_bar, 2);
202 D_t(1,0) = D_t(1,0) - A_i * E_t_i * (y_i - y_bar);
203 D_t(0,1) = D_t(0,1) - A_i * E_t_i * (y_i - y_bar);
204 }
