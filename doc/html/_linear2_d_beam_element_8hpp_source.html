184 void calc_N()
185 {
186 for (int i = 0; i < gauss_points_x.size(); ++i)
187 {
188 N[i](0,0) = 1 - (gauss_points_x[i]/length);
189 N[i](0,3) = gauss_points_x[i] / length;
190 N[i](1,1) = 1 - 3*std::pow(gauss_points_x[i]/length,2) + 2*std::pow(gauss_points_x[i]/length,3);
191 N[i](1,2) = gauss_points_x[i] - 2*std::pow(gauss_points_x[i],2)/length + std::pow(gauss_points_x[i]/length, 2)*gauss_points_x[i];
192 N[i](1,4) = 3*std::pow(gauss_points_x[i]/length, 2) - 2*std::pow(gauss_points_x[i]/length, 3);
193 N[i](1,5) = -gauss_points_x[i]*(gauss_points_x[i]/length) + gauss_points_x[i] * std::pow(gauss_points_x[i]/length,2);
194 }
195 }
200 void calc_B()
201 {
202 for (int i = 0; i < gauss_points_x.size(); ++i)
203 {
204 B[i](0,0) = -1/length;
205 B[i](0,3) = 1/length;
206 B[i](1,1) = -6*std::pow(1/length,2) + 12*gauss_points_x[i]*std::pow(1/length,3);
207 B[i](1,2) = - 4/length + 6*gauss_points_x[i]*std::pow(1/length, 2);
208 B[i](1,4) = 6*std::pow(1/length, 2) - 12*gauss_points_x[i]*std::pow(1/length, 3);
209 B[i](1,5) = -2/length + 6 * gauss_points_x[i]* std::pow(1/length,2);
210 }

260 void update_state()
261 {
262 get_U_from_nodes();
263 // need to retrieve local displacement from the global displacement first of all.
264 calc_d_from_U();
265 // calculating element strain and stress states depends on local displacement d, even though B calculation currently does not.
266 calc_B();
267 calc_eps();
268 calc_stresses();
269 // stiffness calculation must come AFTER stress calculation as stiffness may depend on stress.
270 calc_stiffnesses();
271 // nodal forces depend on stiffness, so must come after stiffness calculations.
272 calc_local_f();
273 // these local element nodal forces are then transformed to global nodal forces.
274 calc_element_global_resistance_forces();
275 // finally, we map back the nodal forces to the force triplets that will be used to populate the global force/resistance vector.
276 populate_resistance_force_triplets();
277 }

289 void calc_mat_stiffness() {
290 real A = section[0]->get_A();
291 real E = section[0]->get_E();
292 real I = section[0]->get_I();
293 real EA = E*A;
294 real EI = E*I;
295 // Row 1
296 local_mat_stiffness(0,0) = EA/length;
297 local_mat_stiffness(0,3) = -EA/length;
298 // Row 2
299 local_mat_stiffness(1,1) = 12*EI/std::pow(length,3);
300 local_mat_stiffness(1,2) = 6*EI/std::pow(length,2);
301 local_mat_stiffness(1,4) = -12*EI/std::pow(length,3);
302 local_mat_stiffness(1,5) = 6*EI/std::pow(length,2);
303 // Row 3
304 local_mat_stiffness(2,1) = 6*EI/std::pow(length,2);
305 local_mat_stiffness(2,2) = 4*EI/length;
306 local_mat_stiffness(2,4) = -6*EI/std::pow(length,2);
307 local_mat_stiffness(2,5) = 2*EI/length;
308 // Row 4
309 local_mat_stiffness(3,0) = -EA/length;
310 local_mat_stiffness(3,3) = EA/length;
311 // Row 5
312 local_mat_stiffness(4,1) = -12*EI/std::pow(length,3);
313 local_mat_stiffness(4,2) = -6*EI/std::pow(length,2);
314 local_mat_stiffness(4,4) = 12*EI/std::pow(length,3);
315 local_mat_stiffness(4,5) = -6*EI/std::pow(length,2);
316 // Row 6
317 local_mat_stiffness(5,1) = 6*EI/std::pow(length,2);
318 local_mat_stiffness(5,2) = 2*EI/length;
319 local_mat_stiffness(5,4) = -6*EI/std::pow(length,2);
320 local_mat_stiffness(5,5) = 4*EI/length;
321 }
