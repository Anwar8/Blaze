55 restraint.apply_restraints(glob_mesh);
56 }
57 #endif
58 // initialise all the loads from the load manager
59 load_manager.initialise_loads();

61

63
64
65 // based on the restraints and loads, we can now initialise the global matrices and establish element-stiffness mapping.
66 // start with mapping stiffnesses as these are needed for initialising the sparse matrices (the stiffness matrix).
