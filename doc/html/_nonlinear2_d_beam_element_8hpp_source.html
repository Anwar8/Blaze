311 void update_state()
312 {
313 get_U_from_nodes();
314 // the corotational transformation actually really cares about the global displacements, so we need to update the global displacements first.
315 transformation.update_state(global_ele_U);
316 // need to retrieve local displacement from the global displacements, which in our case actually uses the transformation object as the relationship is nonlinear!
317 calc_d_from_U();
318 calc_length();
319
320 // calculating element strain, stress, and resistance forces local_f depends on local displacement d in a nonlinear way.
321 calc_eps();
322 calc_stresses();
323 calc_local_f();
324
325 // stiffness calculation must come AFTER stress calculation as stiffness may depend on stress.
326 calc_stiffnesses();
327
328 // these local element nodal forces are transformed to global nodal forces.
329 calc_element_global_resistance_forces();
330 // finally, we map back the nodal forces to the force triplets that will be used to populate the global force/resistance vector.
331 populate_resistance_force_triplets();
332 }

344 void calc_mat_stiffness() {
345 real EA = local_constitutive_mat[0](0,0);
346 real EI = local_constitutive_mat[0](1,1);
347
348 real theta1 = local_d(1);
349 real theta2 = local_d(2);
350 vec V = make_xd_vec(3);
351
352 V(0) = 1/initial_length;
353 V(1) = 2*theta1/15 - theta2/30;
354 V(2) = -theta1/30 + 2*theta2/15;
355 local_mat_stiffness.setZero();
356
357 local_mat_stiffness(1,1) = 4*EI/initial_length;
358 local_mat_stiffness(2,2) = 4*EI/initial_length;
359 local_mat_stiffness(1,2) = 2*EI/initial_length;
360 local_mat_stiffness(2,1) = 2*EI/initial_length;
361 local_mat_stiffness = local_mat_stiffness + EA*initial_length*V*V.transpose();
362 }
