92 {
93 if (end_coords.size() != 2)

96 exit(1);

99 NodeIdCoordsPairsVector node_map;
100 ElemIdNodeIdPairVector elem_map;
101 node_map.reserve(divisions + 1);
102 elem_map.reserve(divisions);
103
104
105 coords delta_xyz = (end_coords[1] - end_coords[0])/divisions;
106
107 for (size_t i = 0; i < divisions + 1; ++i)
108 {
109 node_map.push_back(std::make_pair(i + 1, end_coords[0] + i*delta_xyz));
110 }
111
112 for (size_t i = 0; i < divisions; ++i)
113 {
114 elem_map.push_back(std::make_pair(i + 1, std::vector<size_t>{i + 1, i + 2}));
115 }
116 return std::make_pair(node_map, elem_map);
117 }
