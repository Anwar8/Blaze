125 real F = local_f(0);

129 real L0 = corot_transform.get_L0();
130 local_f(0) = EA*((delta/L0) + (2*theta1*theta1 - theta1*theta2 + 2*theta2*theta2)/30);
131 local_f(1) = ((4*EI/L0) + (2*F*L0/15))*theta1 + ((2*EI/L0) - F*L0/30)*theta2;
