353 void update_state()
354 {
355 get_U_from_nodes();
356 // the corotational transformation actually really cares about the global displacements, so we need to update the global displacements first.
357 transformation.update_state(global_ele_U);
358 // need to retrieve local displacement from the global displacements, which in our case actually uses the transformation object as the relationship is nonlinear!
359 calc_d_from_U();
360 calc_length();
361
362 // calculating element strain, stress, and resistance forces local_f depends on local displacement d in a nonlinear way.
363 calc_B();
364 calc_eps();
365 calc_local_constitutive_mat(); // need to update the constitutive matrix each iteration due to material nonlinearity.
366 calc_stresses();
367 calc_local_f(); // numerical integration required to calculate local_f.
368
369 // stiffness calculation must come AFTER stress calculation as stiffness may depend on stress.
370 calc_stiffnesses();
371
372 // these local element nodal forces are transformed to global nodal forces.
373 calc_element_global_resistance_forces();
374 // finally, we map back the nodal forces to the force triplets that will be used to populate the global force/resistance vector.
375 populate_resistance_force_triplets();
376 }
