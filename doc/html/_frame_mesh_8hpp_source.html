356 std::vector<std::pair<unsigned, std::vector<unsigned>>> map_elements_to_nodes()
357 {
358 std::vector<std::pair<unsigned, std::vector<unsigned>>> elements_map;
359 elements_map.reserve(nbays * nfloors * beam_divisions + (nbays + 1)*nfloors*column_divisions);
360 unsigned element_id = 0;
361
362 for (int bay_i = 1; bay_i <= nbays; ++bay_i) {
363 // first column line of each bay
364 std::set<unsigned> node_ids = get_column_line_node_ids(bay_i - 1);
365 auto it = node_ids.begin();
366 for (unsigned node_id_i = 0; node_id_i < node_ids.size() - 1; ++node_id_i) {
367 element_id++;
368 elements_map.push_back(std::make_pair(element_id, std::vector<unsigned>{*it, *(++it)}));
369 }
370 // beams on each floor
371 for (int floor_i = 1; floor_i <= nfloors; ++floor_i) {
372 node_ids = get_beam_node_ids(bay_i, floor_i, true);
373 it = node_ids.begin();
374 for (unsigned node_id_i = 0; node_id_i < node_ids.size() - 1; ++node_id_i) {
375 element_id++;
376 elements_map.push_back(std::make_pair(element_id, std::vector<unsigned>{*it, *(++it)}));
377 }
378 }
379 }
380 // last column line
381 std::set<unsigned> node_ids = get_column_line_node_ids(nbays);
382 auto it = node_ids.begin();
383 for (unsigned node_id_i = 0; node_id_i < node_ids.size() - 1; ++node_id_i) {
384 element_id++;
385 elements_map.push_back(std::make_pair(element_id, std::vector<unsigned>{*it, *(++it)}));
386 }
