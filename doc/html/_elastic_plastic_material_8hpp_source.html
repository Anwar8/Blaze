61 virtual void increment_strain(real d_eps)
62 {
63
64 initialise_current_state(); // Copy the starting_ state variables into the current state variables
65 real delta_s = E * d_eps; // calculate elastic stress increment
66 check_if_loading(delta_s); // evaluate loading vs unloading/reloading states
67 real s = stress + delta_s; // estimate stress if entire step is elastic
68 evolve_yield_surface(); // perform the hardening calculation
69 strain += d_eps; // increment the strain by the strain increment
70
71
72 if (elastic) // if the material starts off in its iteration as elastic
73 {
74 // we check if it will remain elastic in this step, first
75 eval_yield_function(s); // updates the 'elastic' variable based on incremented stress
76
77 if (elastic) // this means the material did not yield in this load-step
78 {
79 stress = s; // so everything is just elastic - just update the stress and exit this function.
80 return;
81 } else { // the material did, in fact, yield during this step.
82 E_t = E * H/ (E + H); // update the tangent modulus as per Bhatti's similar-triangles approach
83 beta = (fy_bar - abs(stress))/(abs(s) - abs(stress)); // calculate the value of \f$\beta\f$ which is non-zero
84 }
85 } else {
86 // we only enter here if we are plastic to begin with
87 if (loading) // if material is plastic and we are loading, then we continue being plastic - not very interesting.
88 {
89 beta = 0;
90 } else { // otherwise, material is unloading and is thus elastic again
91 elastic = true;
92 E_t = E;
93 stress = s;
94 return;
95 }
96 }
97 stress = stress + beta*delta_s + ((E * H)/(E + H)) * (1 - beta) * d_eps;
98 calc_plastic_flow(d_eps); // increment the plastic strain
99 }
