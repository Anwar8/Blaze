191 virtual void calc_global_stiffness_triplets() override
192 {
193 this->global_stiffness_triplets.clear();
194 // we have the same number of contribution as stiffness components
195 // assuming all are non-zero, which is not correct but okay as it is "safe" although not very memory efficient.
196 this->global_stiffness_triplets.reserve(this->stiffness_map.size());
197 for (auto& kmap: this->stiffness_map)
198 {
199 real val = this->elem_global_stiffness(kmap[0], kmap[1]);
200 this->global_stiffness_triplets.push_back(spnz(kmap[2], kmap[3], val));
201 }

210 virtual void map_stiffness() override
211 {
212 std::vector<int> local_position_vector_rows;
213 std::vector<int> local_position_vector_columns;
214 std::vector<int> position_vector_rows;
215 std::vector<int> position_vector_columns;
216
217 std::vector<int> previous_nodes_ndofs;
218 int element_ndofs = 0;
219 for (auto& node: this->nodes)
220 {
221 previous_nodes_ndofs.push_back(element_ndofs);
222 element_ndofs += node->get_ndof();
223 }
224
225 // set the elements of the position_vector_for_rows
226 for (auto& node: this->nodes)
227 {
228 std::vector<int> node_dofs_numbers;
229 if (node->is_on_parent_rank())
230 {
231 node_dofs_numbers = node->get_dofs_numbers();
232 position_vector_rows.insert(position_vector_rows.end(), node_dofs_numbers.begin(), node_dofs_numbers.end());
233 }
234 }
235 // set the elements of the position_vector_for_columns
236 for (auto& node: this->nodes)
237 {
238 std::vector<int> node_dofs_numbers;
239 node_dofs_numbers = node->get_dofs_numbers();
240 position_vector_columns.insert(position_vector_columns.end(), node_dofs_numbers.begin(), node_dofs_numbers.end());
241 }
242
243 // set the elements of the local_position_vector_for_rows
244 for (int node_i = 0; node_i < this->nnodes; ++node_i)
245 {
246 std::vector<int> node_active_dofs;
247 std::set<int> node_active_dofs_set;
248 if ( this->nodes[node_i]->is_on_parent_rank())
249 {
250 node_active_dofs_set = this->nodes[node_i]->get_active_dofs();
251 node_active_dofs = std::vector<int>(node_active_dofs_set.begin(), node_active_dofs_set.end());
252 std::for_each(node_active_dofs.begin(), node_active_dofs.end(), [=](int& active_dof) { active_dof+=node_i*6;});
253 local_position_vector_rows.insert(local_position_vector_rows.end(), node_active_dofs.begin(), node_active_dofs.end());
254 }
255 }
256
257 // set the elements of the local_position_vector_for_columns
258 for (int node_i = 0; node_i < this->nnodes; ++node_i)
259 {
260 std::vector<int> node_active_dofs;
261 std::set<int> node_active_dofs_set;
262 node_active_dofs_set = this->nodes[node_i]->get_active_dofs();
263 node_active_dofs = std::vector<int>(node_active_dofs_set.begin(), node_active_dofs_set.end());
264 std::for_each(node_active_dofs.begin(), node_active_dofs.end(), [=](int& active_dof) { active_dof+=node_i*6;});
265 local_position_vector_columns.insert(local_position_vector_columns.end(), node_active_dofs.begin(), node_active_dofs.end());
266 }
267
268 this->stiffness_map.clear();
269 int stiffness_size = local_position_vector_rows.size() * local_position_vector_columns.size();
270 this->stiffness_map.reserve(stiffness_size);
271
272 for (int row_i = 0; row_i < local_position_vector_rows.size(); ++row_i)
273 {
274 for (int column_i = 0; column_i < local_position_vector_columns.size(); ++column_i)
275 {
276 this->stiffness_map.push_back({local_position_vector_rows[row_i], local_position_vector_columns[column_i], position_vector_rows[row_i], position_vector_columns[column_i]});
277 }
278 }
279
280 this->resistance_map.clear();
281 int resistance_size = local_position_vector_rows.size();
282 this->resistance_map.reserve(resistance_size);
283
284 for (int row_i = 0; row_i < local_position_vector_rows.size(); ++row_i)
285 {
286 this->resistance_map.push_back({local_position_vector_rows[row_i],position_vector_rows[row_i]});
287 }
