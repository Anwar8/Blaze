401 void populate_resistance_force_triplets() {
402 global_R_triplets.clear();
403 // the 12x1 full resistance vector from local nodal forces vector f
404 if (VERBOSE)
405 {
406 std::cout << "element " << id << " has untransformed local_f " <<std::endl << local_f << std::endl;
407 }
408 vec element_resistance_forces = orient.get_T().transpose()*local_f;
409 if (VERBOSE)
410 {
411 std::cout << "element " << id << " has element_resistance_forces " <<std::endl << element_resistance_forces << std::endl;
412 }
413 std::set<int> node_active_dofs;
414 int nz_i = 0;
415 real force_value;
416 int total_nodal_ndofs_completed = 0; // each node we finish with, we add 6 to this.
417 // This means we have to move to the next set of values corresponding to the next
418 // node in the full resistance vector.
419 for (auto node: nodes)
420 {
421 int nodal_dof_index = 0;
422 node_active_dofs = node->get_active_dofs();
423 nz_i = node->get_nz_i();
424 for (auto active_dof: node_active_dofs)
425 {
426
427 force_value = element_resistance_forces(active_dof + total_nodal_ndofs_completed);
428 // since inactive nodes do not appear in R, we have to make sure to be careful about where we add our nodal forces.
429 // here, nz_i + nodal_dof_index simply starts at where the node freedoms start in the global index, and then
430 // iterates one by one. See how we ++ nodal_dof_index for each freedom we add, and how we restrat from zero when
431 // we start work with the next node?
432 global_R_triplets.push_back(spnz(nz_i + nodal_dof_index, 0, force_value));
433 if (VERBOSE)
434 {
435 std::cout << "element " << id << " node " << node->get_id() << " pushed force_value " << force_value << " to " << nz_i + nodal_dof_index << std::endl;
436 }
437 nodal_dof_index++;
438 }
439 //**< has to be 6 because each node has 6 dofs and our \ref element_resistance_forces also has 6 rows for each node!*
440 total_nodal_ndofs_completed += 6;
441 }
442 }
