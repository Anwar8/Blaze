127void set_from_triplets(Teuchos::RCP<TpetraCrsMatrix> A, std::vector<spnz>& triplets)
128{
129 A->resumeFill();
130 // we have repeats in our row values, so we need to sum rather than replace. Therefore, we have to start by first zeroing the entire matrix.
131 A->setAllToScalar(0.0);
132 auto row_map = map_triplets_to_rows(triplets);
133 for (const auto& row_entry : row_map)
134 {
135 global_ordinal_type row = row_entry.first;
136 const std::vector<global_ordinal_type>& cols = row_entry.second.first;
137 const std::vector<scalar_type>& vals = row_entry.second.second;
138
139 // Convert std::vector to Teuchos::ArrayView<const T>
140 Teuchos::ArrayView<const global_ordinal_type> col_view(cols);
141 Teuchos::ArrayView<const scalar_type> val_view(vals);
142
143 // Replace existing values in CrsMatrix
144 A->sumIntoGlobalValues(row, col_view, val_view);
145 }
146 A->fillComplete();
147}
